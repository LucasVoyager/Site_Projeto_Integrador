CREATE TABLE clientes (
  cpf_cnpj          varchar(20)  NOT NULL PRIMARY KEY,
  nome_razao_social varchar(100) NOT NULL,
  nome_fantasia     varchar(100),
  email             varchar(100),
  telefone          varchar(20),
  celular           varchar(20),
  cep               varchar(50),
  logradouro        varchar(100) NOT NULL,
  complemento       varchar(50),
  numero            varchar(10)  NOT NULL,
  bairro            varchar(50)  NOT NULL,
  cidade            varchar(50)  NOT NULL,
  estado            varchar(2)   NOT NULL,
  pais              varchar(50)  NOT NULL
);

CREATE TABLE servicos (
    id INT AUTO_INCREMENT PRIMARY KEY,
    servico VARCHAR(40) NOT NULL
);

CREATE TABLE contas_receber (
    id INT AUTO_INCREMENT PRIMARY KEY,
    cpf_cnpj VARCHAR(20) NOT NULL,
    valor DECIMAL(10, 2) NOT NULL,
    data_emissao DATE NOT NULL,
    data_vencimento DATE NOT NULL,
    servicos_id INT NOT NULL,
    status ENUM('em aberto', 'pago', 'em atraso') DEFAULT 'em aberto',
    FOREIGN KEY (cpf_cnpj) REFERENCES clientes(cpf_cnpj) ON DELETE RESTRICT,
    FOREIGN KEY (servicos_id) REFERENCES servicos(id) ON DELETE RESTRICT
);
